\section{Front-end: Lexer \& Parser}
% ─────────────────────────────────────────────────────────────────────────────

\begin{frame}[fragile]{Lexical Analyzer (Flex)}
\begin{columns}[t]
  \column{0.40\textwidth}
  \begin{block}{Main Token Classes}\small
    \begin{description}
      \item[\textbf{Keywords}] \texttt{if, while, return, int,\ldots}
      \item[\textbf{Identifiers}] \texttt{[a-zA-Z][a-zA-Z0-9]*}
      \item[\textbf{Integers}] \texttt{[0-9]+}
      \item[\textbf{Reals}] \texttt{[0-9]+\textbackslash.[0-9]+}
      \item[\textbf{Operators}] \texttt{==>} before \texttt{==}
      \item[\textbf{Annotations}] \texttt{@pre}, \texttt{@post}, \texttt{@invariant}
    \end{description}
  \end{block}
  \begin{alertblock}{Order Matters}
    \small Flex rules are evaluated in declaration order.
    \texttt{"result"} must precede the \texttt{ID} pattern, and
    \texttt{"==>"}\ must precede \texttt{"=="}.
  \end{alertblock}

  \column{0.50\textwidth}
  \begin{exampleblock}{Special Rules -- Phase 5 (\texttt{lexer.l})}
\begin{lstlisting}[style=cppslide]
/* formal verification tokens */
"result"     { return RESULT_TK; }
"@pre"       { return PRE_TK; }
"@post"      { return POST_TK; }
"@invariant" { return INV_TK; }
"@variant"   { return VAR_TK; }
"==>"        { return IMPLIES_TK; }

/* relational operators */
"=="  { return EQ_TK; }
"!="  { return NEQ_TK; }
"<="  { return LTE_TK; }
">="  { return GTE_TK; }

/* must come AFTER keywords */
[a-zA-Z][a-zA-Z0-9_]* { return ID_TK; }

/* literals */
[0-9]+\.[0-9]+  { return REAL_TK; }
[0-9]+          { return INT_TK; }
\end{lstlisting}
  \end{exampleblock}
\end{columns}
\end{frame}

\section{Semantic Analysis}

\begin{frame}[fragile]{Symbol Table and Type Analysis}
\begin{columns}[t]
  \column{0.45\textwidth}
  \begin{block}{\texttt{SymbolTable} -- Structure}
\begin{lstlisting}[style=cppslide]
struct Symbol {
  string name;
  unsigned type;     // INT=0, REAL=1, BOOL=2
  unsigned addr;     // stack offset (bytes)
  bool isConstant;   // for 'final'
  bool isFunction;
  int  returnType;
  // formal annotations
  ExprNode* precond;
  ExprNode* postcond;
};
class SymbolTable {
  SymbolTable* parent; // parent scope
  map<string,Symbol> table;
public:
  bool    set(Symbol s);  // false if redecl.
  Symbol* get(string n);  // walks scope chain
  SymbolTable* newScope();
};
\end{lstlisting}
  \end{block}

  \column{0.5\textwidth}
  \begin{block}{Semantic Checks}
    \begin{enumerate}\small
      \item Declaration before use
      \item Type compatibility in operations
      \item No reassignment of \texttt{final}
      \item Return type matches method signature
      \item Unreachable code (post-\texttt{return})
      \item Valid function calls
      \item Division by constant zero (warning)
    \end{enumerate}
  \end{block}
  \begin{exampleblock}{Type Coercion Rules}
    \small
    \begin{tabular}{lcl}
      \texttt{int} op \texttt{double} & $\Rightarrow$ & \texttt{double} (\texttt{IR\_ITOR})\\
      \texttt{int} / \texttt{int}     & $\Rightarrow$ & \texttt{int} (integer division)\\
      \texttt{bool} op \texttt{bool}  & $\Rightarrow$ & \texttt{boolean}\\
    \end{tabular}
  \end{exampleblock}
\end{columns}
\end{frame}

\begin{frame}{Three-Address Code -- Design}

  \begin{block}{\texttt{IRQuad} structure: $(op,\;arg_1,\;arg_2,\;res)$}

    \begin{tabular}{ll}

  \begin{block}{Argument Types (\texttt{IRArgKind})}

\begin{frame}[fragile]{IR Optimizations: Three Passes}
\begin{columns}[t]
  \column{0.33\textwidth}
  \begin{block}{1. Dead Code Elimination}
    \small A temporary $t$ is \emph{dead} if defined but never used as an operand.
    Iteration to \textbf{fixed point}:

    \medskip
    \texttt{used} $\leftarrow$ all \texttt{arg1}, \texttt{arg2}\\
    Remove quads with \texttt{result} $\notin$ \texttt{used}\\
    Repeat until no change

    \medskip
    Not removed: \texttt{JMP, LABEL, RETURN,\\PRINT, CALL} (side effects)
  \end{block}

  \column{0.33\textwidth}
  \begin{block}{2. Constant Folding}
    \small If $arg_1$ and $arg_2$ are literals, evaluate at compile time:

    \medskip
    $\texttt{ADD}\;3\;4 \to \texttt{COPY}\;7$\\
    $\texttt{MUL}\;2\;5 \to \texttt{COPY}\;10$\\
    $\texttt{LTE}\;3\;5 \to \texttt{COPY}\;1$\\
    $\texttt{DIV}\;x\;0 \to$ \texttt{error}

    \medskip
    Eliminates computations independent of variables.
  \end{block}

  \column{0.34\textwidth}
  \begin{block}{3. Copy Propagation}
    \small If \texttt{COPY}\;$a$\;$\to$\;$t_i$ exists, replace uses of $t_i$ with $a$:

    \medskip
    \texttt{COPY   s -> t1}\\
    \texttt{ADD    t1 i -> t2} $\Longrightarrow$\\
    \texttt{ADD    s  i -> t2}

    \medskip
    Then \texttt{t1} becomes dead and pass 1 removes it.
  \end{block}
\end{columns}

\vspace{0.3cm}
\begin{beamercolorbox}[rounded=true,center,wd=0.85\textwidth]{block title example}
  \centering\small\bfseries
  Combined result: $-31\%$ instructions in reference tests
\end{beamercolorbox}
\end{frame}

\begin{frame}{Chaitin-Briggs: Phase 1 -- Liveness Analysis}
\begin{columns}[t]
  \column{0.52\textwidth}
  \begin{block}{Fixed-Point Equations per Basic Block $B$}
    \bigskip
    \begin{align*}
      \text{use}[B]        &= \text{vars used \emph{before} being defined in }B\\[6pt]
      \text{def}[B]        &= \text{vars \emph{defined} in }B\\[10pt]
      \text{live\_in}[B]  &= \text{use}[B] \;\cup\;
                              \bigl(\text{live\_out}[B] \setminus \text{def}[B]\bigr)\\[6pt]
      \text{live\_out}[B] &= \bigcup_{S \in \text{succ}(B)} \text{live\_in}[S]
    \end{align*}
    \bigskip
    Backward iteration (successors to predecessors)\\
    until no \texttt{live\_in}/\texttt{live\_out} changes (\textbf{fixed point}).
  \end{block}

  \column{0.44\textwidth}
  \begin{exampleblock}{Example: Loop Block}
    \small
    \begin{tabular}{lll}
      \textbf{Instruction} & \textbf{use} & \textbf{def} \\

      \texttt{COPY t1 -> s}   & $t1$   & $s$  \\
      \texttt{ADD i 1  -> t2} & $i$    & $t2$ \\
      \texttt{COPY t2 -> i}   & $t2$   & $i$  \\
      \texttt{JMP L0}         & ---    & --- \\
    \end{tabular}

    \medskip
    \textbf{live\_out} of \texttt{LTE} instruction:\\
    $\{i,\, s,\, t0\}$
  \end{exampleblock}
  \begin{alertblock}{\small Key Property}
    \small If $d$ is defined with $v \in \text{live\_out}$\\
    $\Rightarrow$ edge $(d, v)$ in the interference graph
  \end{alertblock}

\section{Formal Verification}

\begin{frame}{Hoare Logic: Triple and Verification Method}
\begin{columns}[c]
  \column{0.48\textwidth}
  \begin{block}{Hoare Triple}
    \bigskip
    \[ \hoare{P}{S}{Q} \]
    \bigskip
    \normalsize
    If precondition $P$ holds \emph{before} executing $S$,
    and execution \emph{terminates}, then postcondition $Q$
    holds \emph{after}.
    \bigskip
  \end{block}
  \begin{exampleblock}{Concrete Example}
    \[ \hoare{\mathit{true}}{\texttt{x:=-5; if(x<0) ret -x}}{\texttt{result} \geq 0} \]
    \small Trivial precondition, postcondition: non-negative result.
  \end{exampleblock}

  \column{0.48\textwidth}
  \begin{alertblock}{Main Condition to Verify}
    \bigskip
    \[ P \;\Rightarrow\; \WP{\mathit{body}}{Q} \]
    \bigskip
    \small
    \textbf{valid} $\iff$ Z3 returns \texttt{unsat} to $\neg(P \Rightarrow \WP{\mathit{body}}{Q})$
    \bigskip
  \end{alertblock}
  \begin{block}{Method Rule}
    \bigskip
    For \texttt{@pre}/\texttt{@post} annotations:
    \[ \mathit{VC}_{\mathrm{corr}} \;=\; P \Rightarrow \WP{\mathit{body}}{Q} \]
    \small Generated automatically by \texttt{VCGenerator}.
    \bigskip
  \end{block}
\end{columns}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}{WP Calculus: Rules (1/2) -- Assignment, Sequence, Return}

\vspace{0.2cm}
\begin{columns}[t]
  \column{0.50\textwidth}

  \begin{block}{Assignment}
    \[
      \WP{x := e}{\,Q\,} \;=\; Q\bigl[e/x\bigr]
    \]
    \small
    Textual substitution of $x$ with $e$ in $Q$.
  \end{block}

  \vspace{0.5cm}

  \begin{block}{Sequence}
    \[
      \WP{S_1\,;\,S_2}{\,Q\,} \;=\; \WP{S_1}{\,\WP{S_2}{Q}\,}
    \]
    \small
    \emph{Backward} composition: computed from the end.
  \end{block}

  \vspace{0.5cm}

  \begin{block}{Return}
    \[
      \WP{\texttt{return}\;e}{\,Q\,} \;=\; Q\bigl[e/\texttt{result}\bigr]
    \]
    \small
    Same as assignment, substituting \texttt{result}.
  \end{block}

  \column{0.46\textwidth}

  \begin{exampleblock}{Example -- Assignment}
    \medskip
    \[
      \WP{x := 3}{\;x > 0\;} \;=\; (3 > 0) \;=\; \mathit{true}
    \]
    \medskip
  \end{exampleblock}

  \begin{exampleblock}{Example -- Sequence}
    \medskip
    \small
    $\WP{x:=3\;;\;y:=x+1}{\;y>0\;}$
    \begin{align*}
      &= \WP{x:=3}{\;\WP{y:=x+1}{\;y>0\;}\;}\\
      &= \WP{x:=3}{\;(x+1)>0\;}\\
      &= (3+1) > 0 \;=\; \mathit{true}
    \end{align*}
    \medskip
  \end{exampleblock}

  \begin{exampleblock}{Example -- Return}
    \medskip
    $\WP{\texttt{ret}\;{-x}}{\;\texttt{result}\geq 0\;}$
    \[= (-x) \geq 0\]
    \medskip
  \end{exampleblock}

\end{columns}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}{WP Calculus: Rules (2/2) -- Conditional and Loop}

\vspace{0.2cm}
\begin{columns}[t]
  \column{0.50\textwidth}

  \begin{block}{Conditional}
    \begin{align*}
      &\WP{\texttt{if}(B)\;S_1\;\texttt{else}\;S_2}{\,Q\,}\\[6pt]
      &\quad=\; \bigl(B \Rightarrow \WP{S_1}{Q}\bigr)\\
      &\quad\wedge\; \bigl(\neg B \Rightarrow \WP{S_2}{Q}\bigr)
    \end{align*}
    \small
    Each branch must guarantee $Q$ under its entry condition.
  \end{block}

  \vspace{0.5cm}

  \begin{block}{Loop with Invariant $I$}
    \[
      \WP{\texttt{while}_I(B)\;\{S\}}{\,Q\,} \;=\; I
    \]
    \small
    The WP of the loop \emph{is} its invariant.
    The 4 correctness VCs are generated separately by \texttt{VCGenerator}.
  \end{block}

  \column{0.46\textwidth}

  \begin{exampleblock}{Example -- Conditional}
    \medskip
    \small
    $Q = \texttt{result} \geq 0$, \quad $B = (x < 0)$
    \begin{align*}
      &\WP{\texttt{if}(x\!<\!0)\;\{\texttt{ret}\;{-x}\}\;\texttt{else}\;\{\texttt{ret}\;x\}}{Q}\\[4pt]
      &= (x\!<\!0 \Rightarrow -x \geq 0)\\
      &\quad \wedge\; (x\!\geq\!0 \Rightarrow x \geq 0)
    \end{align*}
    \medskip
  \end{exampleblock}

  \begin{alertblock}{Why the Loop Returns $I$}
    \medskip
    \small
    The compiler cannot compute the WP of a loop without
    knowing how many iterations it will run. The invariant $I$
    acts as a \emph{summary} of the loop's effect.
    \medskip
  \end{alertblock}

  \begin{block}{\small Classic References}
    \tiny
    Dijkstra, \textit{A Discipline of Programming}, 1976\\
    Hoare, \textit{Axiomatic Basis for Computer Programming}, 1969
  \end{block}

\end{columns}
\end{frame}

\begin{frame}{WP Derivation (2/2): Applying the Rules}

\begin{block}{Target postcondition: $Q = \texttt{result} \geq 0$}
\end{block}
\vspace{0.3cm}

\begin{columns}[t]
  \column{0.50\textwidth}

  \textbf{Step 1} -- Return rule:
  \vspace{2pt}
  \begin{align*}
    &\WP{\texttt{ret}\;x}{Q} = Q\bigl[x/\texttt{result}\bigr] = (x \geq 0)\\
    &\WP{\texttt{ret}\;{-x}}{Q} = Q\bigl[-x/\texttt{result}\bigr] = (-x \geq 0)
  \end{align*}

  \vspace{4pt}
  \textbf{Step 2} -- Conditional rule ($B = x < 0$):
  \vspace{2pt}
  \begin{align*}
    \WP{S2}{Q} = \;
      &(x<0) \;\Rightarrow\; (-x \geq 0)\\
      \wedge\;&(x\geq 0) \;\Rightarrow\; (x \geq 0)
  \end{align*}

  \column{0.46\textwidth}

  \textbf{Step 3} -- Assignment rule $x := -5$:
  \vspace{2pt}
  \[
    \WP{x:=-5}{\WP{S2}{Q}} = \WP{S2}{Q}\bigl[-5/x\bigr]
  \]
  \vspace{-4pt}
  \begin{align*}
    = \;&(-5 < 0) \;\Rightarrow\; (-(-5) \geq 0)\\
        &\quad = \mathit{true} \;\Rightarrow\; \mathit{true}
        = \textcolor{accent}{\mathbf{true}}\\[4pt]
    \wedge\;&(-5 \geq 0) \;\Rightarrow\; (-5 \geq 0)\\
        &\quad = \mathit{false} \;\Rightarrow\; \mathit{false}
        = \textcolor{accent}{\mathbf{true}}
  \end{align*}

  \vspace{4pt}
  \textbf{Final VC}: $\;\mathit{true} \Rightarrow
  (\textcolor{accent}{\mathbf{true}} \wedge \textcolor{accent}{\mathbf{true}})$
  \hfill Z3: \texttt{unsat} \ok
\end{columns}
\end{frame}

\begin{frame}[fragile]{The CPS Problem: Early Returns}

\begin{columns}[t]
  \column{0.46\textwidth}
  \begin{alertblock}{Naive Rule: Backward Composition \textbf{Fails}}
    \medskip
    For body \;\texttt{if(x<0) \{ret -x;\} ret x;}:

    \medskip
    The standard rule $\WP{S_1}{\WP{S_2}{Q}}$ processes\\
    \texttt{ret x} ($=S_2$) first $\to$ $Q[x/\texttt{result}]$,\\
    then the \texttt{if} with that postcondition.

    \medskip
    But the \texttt{then} branch executes \texttt{ret -x} and \textbf{exits}: $S_2$
    must not run after it.
    Backward composition silently ignores this.
    \medskip
  \end{alertblock}

  \column{0.50\textwidth}
  \begin{exampleblock}{Solution: \texttt{wpSeq} Forward with Continuation}
\begin{lstlisting}[style=cppslide,basicstyle=\ttfamily\scriptsize]
Predicate* wpSeq(stmts, idx, Q) {
  auto* s = stmts[idx];

  // return: applies Q and DISCARDS the rest
  if (auto* r = cast<ReturnNode>(s))
      return wpReturn(r, Q);  // Q[e/result]

  // if: forks continuation into both branches
  if (auto* i = cast<IfNode>(s)) {
      auto cont = stmts[idx+1 .. end];
      Predicate* B = fromExpr(i->cond);
      return new AndPred(
        new ImpliesPred(B->clone(),
          wpBranchAndCont(i->then, cont, Q)),
        new ImpliesPred(new NotPred(B),
          wpBranchAndCont(i->else_, cont, Q))
      );
  }
  // assign: WP(x:=e; rest, Q) = WP(rest,Q)[e/x]
  if (auto* a = cast<AssignNode>(s))
      return wpAssign(a, wpSeq(stmts,idx+1,Q));

  return wpSeq(stmts, idx+1, Q);
}
\end{lstlisting}
  \end{exampleblock}
\end{columns}
\end{frame}

\begin{frame}{Loop VCs: The Four Conditions}

\begin{center}
  For \;\texttt{while($B$)\{$S$\}}\; with invariant $I$ and variant $V$:
\end{center}

\vspace{0.4cm}
\begin{columns}[t]
  \column{0.50\textwidth}

  \begin{block}{$\mathit{VC}_1$ -- Initiation}
    \[ P \;\Rightarrow\; \WP{S_{\text{preloop}}}{I} \]
    \small The precondition implies the invariant holds \emph{on entry}.
  \end{block}

  \vspace{0.3cm}
  \begin{block}{$\mathit{VC}_2$ -- Preservation}
    \[ (I \wedge B) \;\Rightarrow\; \WP{S_{\text{body}}}{I} \]
    \small If the invariant holds and the guard is true, it still holds after the body.
  \end{block}

  \column{0.50\textwidth}

  \begin{block}{$\mathit{VC}_3$ -- Use of Invariant}
    \[ (I \wedge \neg B) \;\Rightarrow\; \WP{S_{\text{post}}}{Q} \]
    \small On exit ($\neg B$ holds), the invariant implies the postcondition.
  \end{block}

  \vspace{0.3cm}
  \begin{block}{$\mathit{VC}_4$ -- Termination (\texttt{@variant})}
    \[ (I \wedge B) \;\Rightarrow\; \WP{S_{\text{body}}}{0 \leq V' < V} \]
    \small The variant $V$ decreases strictly and stays non-negative.
  \end{block}

\end{columns}
\end{frame}

\begin{frame}{Loop VCs: Strong Invariant and Variant}

\begin{columns}[t]

  \begin{alertblock}{Weak Invariant $\Rightarrow$ VC Fails}

    \medskip

    \medskip

    \medskip
    With $I = (s \geq 0) \wedge (i \geq 1)$:

    \medskip
    $\mathit{VC}_2$: $(s\!\geq\!0 \wedge i\!\geq\!1 \wedge i\!\leq\!5) \Rightarrow (s+i) \geq 0$

    \medskip

    \vspace{0.3cm}

    \medskip
